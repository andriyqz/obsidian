### Оператори порівняння як універсальні функції
```python
x = np.arange(10)

print(x > 3)
print(x < 3)
print(x == 3)
print(x != 3)
print((2**x) == x**2, '\n')

print(np.greater(x, 3))
print(np.less(x, 3))
print(np.equal(x, 3))
print(np.not_equal(x, 3))
print(np.less_equal(x, 3))
print(np.greater_equal(x, 3))
```

```python
rng = np.random.RandomState(0)

x = rng.randint(10, size=(3, 4))

print(np.count_nonzero(x < 6))
print(np.sum(x < 6))
```
