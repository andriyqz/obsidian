### .sort, .argsort
```python
x = np.array([2, 1, 4, 3, 5])
np.sort(x)
x.sort() # changes array
print(x)
```

```python
x = np.array([2, 1, 4, 3, 5])
print(np.argsort(x)) # індекси відсортованого масиву
x = x[np.argsort(x)]
x
```

```python
rand = np.random.RandomState(42)

X = rand.randint(0, 10, (4, 6))

print(np.sort(X, axis=0))
print(np.sort(X)) # default axis=-1
```

### .partition, .argpartition
```python
x = np.array([7, 2, 3, 1, 6, 5, 4])

print(np.partition(x, 4)) # перед 4 індексом будуть елементи <= x[4], після 4 індексу >= x[4]

print(np.argpartition(x, 4)) # перед 4 індексом будуть елементи <= x[4], після 4 індексу >= x[4])
```

### K-NN
```python
X = rand.rand(10, 2)
K = 2

import matplotlib.pyplot as plt

plt.scatter(X[:, 0], X[:, 1])

dist_sq = np.sum((X[np.newaxis, :, :] - X[:, np.newaxis, :])**2)
nearest = np.argpartition(dist_sq, K+1, axis=1)

for i in range(X.shape[0]):
	for j in nearest[i, :K+1]:
		plt.plot(*zip(X[j], X[i]), color='black')
```
