### Арифметичні операції над масивами
```python
x = np.arange(4)

print(x)
print(x + 5)
print(x - 5)
print(x * 2)
print(x / 2)
print(x // 2)

print(np.add(x, 2))
print(np.subtract(x, 2))
print(np.multiply(x, 2))
print(np.divide(x, 2))
print(np.floor_divide(x, 2))
print(np.power(x, 2))
print(np.mod(x, 2))
print(np.negative(x))
```

### Тригонометричні функції
```python
theta = np.linspace(0, np.pi, 3)

print(theta)
print(np.sin(theta))
print(np.cos(theta))
print(np.tan(theta))

x = [-1, 0, 1]
print(np.arcsin(x)) # radians
print(np.arccos(x))
print(np.arctan(x))
```

### Показникові функції і логарифми
```python
x = [1, 2, 3]
print(np.exp(x))
print(np.exp2(x)) # 2^x
print(np.power(3, x)) # 3^x

x = [1, 2, 4, 10]
print(np.log(x)) # ln(x)
print(np.log2(x))
print(np.log10(x))
print(np.expm1(x)) # exp(x) - 1 with higher accuracy with very little input data
print(np.log1p(x)) # log(1+x)
```
